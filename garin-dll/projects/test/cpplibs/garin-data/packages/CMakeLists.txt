    "../../garin-scene/src/*.cpp"
    "../../garin-ui/src/*.cpp"

    "../../garin-vr/src/*.cpp"
    "../../garin-graphics/src/*.cpp"
    "../../garin-maths/src/*.cpp"

    "../../garin-graphics/libs/*.lib"

    "../../garin-audio/libs/*.lib"
)

# Definiciones de compilaci√≥n
